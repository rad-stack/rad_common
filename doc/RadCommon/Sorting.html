def apply_sorting(results)
  if sort_column.present? && sort_direction.present?
    sort_query(results)
  else
    results
  end
end

def sort_clause
  @sort_column.split(',').map { |item| item.strip + ' ' + @sort_direction + ' NULLS LAST' }.join(', ')
end
